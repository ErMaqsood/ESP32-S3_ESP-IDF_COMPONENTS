55struct kalman_motion_t {
56 float q_angle; // Process noise variance for the accelerometer
57 float q_bias; // Process noise variance for the gyro bias
58 float r_measure; // Measurement noise variance - this is actually the variance of the measurement noise
59 float angle; // The angle calculated by the Kalman filter - part of the 2x1 state vector
60 float bias; // The gyro bias calculated by the Kalman filter - part of the 2x1 state vector
61 float rate; // Unbiased rate calculated from the rate and the calculated bias - you have to call getAngle to update the rate
62 float p[2][2]; // Error covariance matrix - This is a 2x2 matrix
63};
